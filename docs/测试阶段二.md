```typescript
// Test: 完整流程
const runFullFlowTest = async () => {
  // 1. Create knowledge base
  const kbResponse = await fetch('/api/knowledge-bases?appId=test-app', {
    method: 'POST',
    body: JSON.stringify({ name: '测试知识库' }),
  });
  expect(kbResponse.status).toBe(201);
  const kb = await kbResponse.json();
  
  // 2. Upload document
  const formData = new FormData();
  formData.append('file', new File(['测试内容'], 'test.txt'));
  const uploadResponse = await fetch(`/api/knowledge-bases/${kb.id}/documents`, {
    method: 'POST',
    body: formData,
  });
  expect(uploadResponse.status).toBe(201);
  
  // 3. Create workflow with RAG node
  const workflowResponse = await fetch('/api/workflows', {
    method: 'POST',
    body: JSON.stringify({
      name: 'RAG 测试工作流',
      appId: 'test-app',
      definition: {
        nodes: [
          { id: 'start', type: 'start' },
          { id: 'rag', type: 'rag', data: { config: { knowledgeBaseId: kb.id } } },
        ],
        edges: [{ source: 'start', target: 'rag' }],
      },
    }),
  });
  expect(workflowResponse.status).toBe(201);
  const workflow = await workflowResponse.json();
  
  // 4. Execute workflow
  const executeResponse = await fetch(`/api/workflows/${workflow.id}/run`, {
    method: 'POST',
    body: JSON.stringify({ input: { query: '测试查询' } }),
  });
  expect(executeResponse.status).toBe(201);
  const result = await executeResponse.json();
  expect(result.status).toBe('success');
};
```

describe('RagExecutor', () => {
  let executor: RagExecutor;
  
  beforeEach(() => {
    executor = new RagExecutor();
  });
  
  it('should execute RAG node', async () => {
    const config: RAGNodeConfig = {
      knowledgeBaseId: 'test-kb',
      query: '测试查询',
      topK: 5,
      similarityThreshold: 0.3,
      outputFormat: 'combined',
    };
    
    const context: ExecutionContext = {
      workflowId: 'test-workflow',
      runId: 'test-run',
      variables: {},
      nodeOutputs: {},
      startTime: new Date(),
    };
    
    const result = await executor.execute(config, context);
    
    expect(result.success).toBe(true);
    expect(result.output).toBeDefined();
  });
  
  it('should fail without knowledgeBaseId', async () => {
    const config: any = { query: 'test' };
    const result = await executor.execute(config, {} as any);
    
    expect(result.success).toBe(false);
    expect(result.error).toContain('knowledgeBaseId');
  });
});
