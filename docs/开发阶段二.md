```prisma
model KnowledgeBase {
  id               String   @id @default(uuid())
  appId            String
  name             String
  description      String?
  embeddingModelId String?
  createdAt        DateTime @default(now())
  updatedAt        DateTime @updatedAt

  documents Document[]

  @@index([appId])

/// 文档表
model Document {
  id             String   @id @default(uuid())
  knowledgeBaseId String
  title          String
  content        String?
  filePath       String?
  fileType       String   // pdf, docx, txt, markdown
  wordCount      Int      @default(0)
  status         String   @default("pending") // pending, indexing, ready, failed
  metadata       Json     @default("{}")
  createdAt      DateTime @default(now())

  knowledgeBase KnowledgeBase @relation(fields: [knowledgeBaseId], references: [id], onDelete: Cascade)
  chunks        DocumentChunk[]

  @@index([knowledgeBaseId])
  @@map("documents")
}

/// 文档分段表（带向量）
model DocumentChunk {
  id         String   @id @default(uuid())
  documentId String
  content    String
  embedding  Unsupported("vector(1536)")? // pgvector
  metadata   Json     @default("{}")
  chunkIndex Int
  createdAt  DateTime @default(now())

  document Document @relation(fields: [documentId], references: [id], onDelete: Cascade)

  @@index([documentId])
  @@map("document_chunks")
}
```

## 🤖 第三部分：Agent 模式（1.5-2 周）

### 2.3.1 核心架构

```
┌─────────────────────────────────────────────┐
│            Agent 引擎                        │
├─────────────────────────────────────────────┤
│  1. 接收用户任务                              │
│  2. 思考（Thought）: 分析任务、制定计划        │
│  3. 行动（Action）: 选择工具、执行             │
│  4. 观察（Observation）: 获取工具返回结果      │
│  5. 循环 2-4 直到任务完成                      │
│  6. 返回最终答案                              │
└─────────────────────────────────────────────┘
```

### 2.6.3 一键部署脚本

**deploy.sh**:

```bash
#!/bin/bash

set -e

echo "🚀 Starting AI Engine deployment..."

# Check if Docker is running
if ! docker info > /dev/null 2>&1; then
  echo "❌ Docker is not running. Please start Docker first."
  exit 1
fi

# Create .env if not exists
if [ ! -f .env ]; then
  echo "📝 Creating .env file..."
  cp .env.example .env
fi

# Start database services
echo "📦 Starting database services..."
docker-compose up -d postgres redis

# Wait for PostgreSQL to be ready
echo "⏳ Waiting for PostgreSQL to be ready..."
sleep 10

# Run database migrations
echo "🔧 Running database migrations..."
cd docker && docker-compose exec -T server pnpm --filter @ai-engine/server db:migrate || true

# Build and start all services
echo "🏗️  Building and starting all services..."
docker-compose up -d --build

# Show status
echo ""
echo "✅ Deployment complete!"
echo ""
echo "📍 Web UI: http://localhost:3001"
echo "📍 API: http://localhost:3000"
echo "📍 Swagger: http://localhost:3000/docs"
echo ""
echo "📊 View logs: docker-compose logs -f"
echo "🛑 Stop all: docker-compose down"
```
