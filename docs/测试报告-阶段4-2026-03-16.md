describe('WorkflowExecutor', () => {
  let executor: WorkflowExecutor
  let mockContext: ExecutionContext

  beforeEach(() => {
    mockContext = {
      workflowId: 'test-workflow',
      runId: 'test-run',
      variables: {},
      nodeOutputs: {},
      startTime: new Date(),
    }
    executor = new WorkflowExecutor()
  })

  describe('execute', () => {
    it('should execute complete workflow', async () => {
      const definition: WorkflowDefinition = {
        nodes: [
          { id: 'start', type: 'start', config: {} },
          { id: 'llm', type: 'llm', config: {} },
          { id: 'end', type: 'end', config: {} },
        ],
        edges: [
          { id: 'e1', source: 'start', target: 'llm' },
          { id: 'e2', source: 'llm', target: 'end' },
        ],
      }

      const result = await executor.execute(definition, { input: 'test' })

      expect(result.status).toBe('success')
      expect(result.nodeResults).toHaveLength(3)
    })

    it('should handle condition branches', async () => {
      const definition: WorkflowDefinition = {
        nodes: [
          { id: 'start', type: 'start', config: {} },
          { id: 'cond', type: 'condition', config: { expression: 'result > 5' } },
          { id: 'pass', type: 'llm', config: {} },
          { id: 'fail', type: 'llm', config: {} },
        ],
        edges: [
          { id: 'e1', source: 'start', target: 'cond' },
          { id: 'e2', source: 'cond', target: 'pass', condition: 'result > 5' },
          { id: 'e3', source: 'cond', target: 'fail', condition: 'result <= 5' },
        ],
      }

      mockContext.variables = { result: 10 }
      const result = await executor.execute(definition, {})

      expect(result.nodeResults.some(n => n.nodeId === 'pass')).toBe(true)
      expect(result.nodeResults.some(n => n.nodeId === 'fail')).toBe(false)
    })
  })

  describe('findNextNode', () => {
    it('should return target for single edge', () => {
      const edges = [{ id: 'e1', source: 'node1', target: 'node2' }]
      const nextNode = (executor as any).findNextNode({ edges }, 'node1', {})
      expect(nextNode).toBe('node2')
    })

    it('should evaluate conditions for multiple edges', () => {
      const edges = [
        { id: 'e1', source: 'node1', target: 'node2', condition: 'result > 5' },
        { id: 'e2', source: 'node1', target: 'node3', condition: 'result <= 5' },
      ]
      const nextNode = (executor as any).findNextNode(
        { edges },
        'node1',
        { result: 10 }
      )
      expect(nextNode).toBe('node2')
    })
  })
})

## 四、问题与解决

### 问题 1: Workflow Executor 依赖 mock
**现象**: Node Executors 已测试，但 Workflow Executor 需要 mock 它们  
**根因**: Workflow Executor 依赖所有 Node Executors，直接测试会调用真实实现  
**解决**: 在测试文件中 mock 所有 Node Executors
```typescript
vi.mock('./node-executors', () => ({
  StartNodeExecutor: vi.fn().mockImplementation(() => ({
    execute: vi.fn().mockResolvedValue({ success: true, output: {} }),
    canExecute: vi.fn(() => true),
  })),
  // ... 其他 executors
}))
```

### 问题 2: vitest.config.ts 路径深度错误
**现象**: `@ai-engine/providers` 导入失败  
**根因**: 路径解析错误，`../packages` 相对于 `apps/server` 应为 `../../packages`  
**解决**: 修复 vitest.config.ts 路径别名

### 问题 3: $transaction 接收数组参数
**现象**: `conversation.service.spec.ts` 测试失败  
**根因**: Prisma 的 `$transaction` 接收数组参数，而非回调函数  
**解决**: 修改 mock 实现支持数组参数

### 问题 4: vi.mock hoisting 问题
**现象**: `chat.service.spec.ts` 中 `mockProvider is not defined`  
**根因**: vi.mock factory 在文件顶部执行，无法访问后面定义的变量  
**解决**: 使用 `vi.hoisted()` 包装 vi.mock，确保 hoisting 正确
