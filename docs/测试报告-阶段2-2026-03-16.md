```typescript
// 修复前：先检查 colon 格式，导致 ollama/llama2:7b 返回 7b
getModelName(modelId: string): string {
  const colonMatch = modelId.match(/^([^:]+):(.+)$/)
  if (colonMatch) return colonMatch[2]
  
  const slashMatch = modelId.match(/^([^/]+)\/(.+)$/)
  if (slashMatch) return slashMatch[2]
  
  return modelId
}

// 修复后：先检查 slash 格式
getModelName(modelId: string): string {
  // Check slash format first (e.g., "ollama/llama2:7b" -> "llama2:7b")
  const slashMatch = modelId.match(/^([^/]+)\/(.+)$/)
  if (slashMatch) return slashMatch[2]
  
  // Then check colon format (e.g., "aliyun:qwen-turbo" -> "qwen-turbo")
  const colonMatch = modelId.match(/^([^:]+):(.+)$/)
  if (colonMatch) return colonMatch[2]
  
  return modelId
}
```

## 四、问题与解决

### 问题 1: getModelName() 逻辑错误
**现象**: `getModelName('ollama/llama2:7b')` 返回 `'7b'` 而非 `'llama2:7b'`  
**根因**: 正则表达式检查顺序错误，先检查了 colon 格式  
**解决**: 调整顺序，先检查 slash 格式，再检查 colon 格式  
**影响**: 修复后模型路由功能正常工作

### 问题 2: 测试环境变量干扰
**现象**: `should return unhealthy for Aliyun without API key` 失败  
**根因**: 开发环境已配置 `ALIYUN_API_KEY`，导致 healthCheck 返回 healthy  
**解决**: 在测试中临时删除 `delete process.env.ALIYUN_API_KEY`  
**影响**: 测试更准确反映真实逻辑

### 问题 3: 对象比较使用 toBe
**现象**: `should cache health check results` 失败  
**根因**: `toBe` 使用引用比较，两次 healthCheck 返回不同对象实例  
**解决**: 改用 `toStrictEqual` 进行深度比较  
**影响**: 测试断言更准确
