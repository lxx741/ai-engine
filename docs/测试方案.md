```yaml
# .github/workflows/test.yml
name: Tests

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    
    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: aiengine123
          POSTGRES_USER: aiengine
          POSTGRES_DB: aiengine
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
    
    steps:
      - uses: actions/checkout@v4
      
      - uses: pnpm/action-setup@v2
        with:
          version: 8
      
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'pnpm'
      
      - name: Install dependencies
        run: pnpm install
      
      - name: Generate Prisma Client
        run: pnpm db:generate
      
      - name: Run unit tests
        run: pnpm test
      
      - name: Run E2E tests
        run: pnpm --filter @ai-engine/server test:e2e
        env:
          DATABASE_URL: postgresql://aiengine:aiengine123@localhost:5432/aiengine
      
      - name: Upload coverage
        run: pnpm --filter @ai-engine/server test:cov
      
      - uses: codecov/codecov-action@v3

          files: ./apps/server/coverage/coverage-final.json
```

### 场景 5: 条件分支 ⭐⭐

**目标**: 验证条件判断功能

**步骤**:
1. 创建工作流
2. 添加节点：
   - start（设置变量 score=85）
   - condition（表达式：{{ score }} >= 60）
   - llm（pass 分支：恭喜通过）
   - llm（fail 分支：未通过）
   - end
3. 配置边：
   - condition → pass（条件：output.result == true）
   - condition → fail（条件：output.result == false）

## 十二、维护说明

### 测试更新规范

1. **新增功能必须附带测试**
   - 新功能 PR 需包含对应测试
   - 测试覆盖率不下降

2. **测试命名规范**
   ```typescript
   describe('ServiceName', () => {
     describe('methodName', () => {
       it('should do something when condition', () => {
         // 测试实现
       });
     });
   });
   ```

3. **Mock 规范**
   - 使用 `jest.fn()` 创建 mock
   - 复杂对象使用 `jest.mock()`
   - 外部服务使用 Test Containers

4. **测试数据管理**
   - 使用 factory 函数生成测试数据
   - 每个测试独立，不依赖顺序
   - 测试后清理数据

## 十、Vitest 迁移总结

### 迁移步骤

1. **安装依赖**
   ```bash
   pnpm --filter @ai-engine/server add -D vitest @vitest/coverage-v8
   pnpm --filter @ai-engine/core add -D vitest @vitest/coverage-v8
   ```

2. **创建 vitest.config.ts**
   ```typescript
   import { defineConfig } from 'vitest/config'
   import path from 'path'

   export default defineConfig({
     test: {
       globals: true,
       environment: 'node',
       include: ['src/**/*.{test,spec}.ts'],
       coverage: {
         provider: 'v8',
         reporter: ['text', 'json', 'html'],
       },
     },
     resolve: {
       alias: {
         '@ai-engine/shared': path.resolve(__dirname, '../shared/src'),
         '@ai-engine/core': path.resolve(__dirname, '../core/src'),
         '@ai-engine/providers': path.resolve(__dirname, '../providers/src'),
       },
     },
   })
   ```

3. **更新 package.json**
   ```json
   {
     "scripts": {
       "test": "vitest run",
       "test:watch": "vitest",
       "test:cov": "vitest run --coverage"
     }
   }
   ```

4. **修改测试文件**
   ```typescript
   // 从 vitest 导入 vi
   import { vi, beforeEach, describe, it, expect } from 'vitest'
   
   // 替换 jest.fn() → vi.fn()
   // 替换 jest.mock() → vi.mock()
   ```
