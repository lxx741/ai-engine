```typescript
// apps/web/src/components/workflow/canvas-provider.ts
import { create } from 'zustand'

interface CanvasState {
  nodes: WorkflowNode[]
  edges: WorkflowEdge[]
  selectedNode?: string
  viewport: { x: number; y: number; zoom: number }
  
  // Actions
  addNode: (node: WorkflowNode) => void
  updateNode: (id: string, config: any) => void
  deleteNode: (id: string) => void
  addEdge: (edge: WorkflowEdge) => void
  deleteEdge: (id: string) => void
  setSelectedNode: (id: string) => void
  saveToLocalStorage: () => void
  loadFromLocalStorage: () => void
}

export const useCanvasStore = create<CanvasState>((set, get) => ({
  nodes: [],
  edges: [],
  viewport: { x: 0, y: 0, zoom: 1 },
  
  addNode: (node) => set((state) => ({ 
    nodes: [...state.nodes, node] 
  })),
  
  updateNode: (id, config) => set((state) => ({
    nodes: state.nodes.map(n => 
      n.id === id ? { ...n, config } : n
    )
  })),
  
  deleteNode: (id) => set((state) => ({
    nodes: state.nodes.filter(n => n.id !== id),
    edges: state.edges.filter(e => 
      e.source !== id && e.target !== id
    )
  })),
  
  // ... 其他 actions
}))
```

```typescript
// packages/core/src/validators/flow-validator.ts
interface NodeIO {
  inputs: string[]
  outputs: string[]
}

const nodeIOConfig: Record<WorkflowNodeType, NodeIO> = {
  start: { inputs: [], outputs: ['output'] },
  llm: { inputs: ['prompt'], outputs: ['response'] },
  http: { inputs: ['url', 'method'], outputs: ['response'] },
  condition: { inputs: ['expression'], outputs: ['true', 'false'] },
  end: { inputs: ['result'], outputs: [] },
  tool: { inputs: ['params'], outputs: ['result'] },
}

export function validateFlow(nodes: WorkflowNode[], edges: WorkflowEdge[]): ValidationResult {
  const errors: string[] = []
  
  // 检查每个节点的输入是否满足
  for (const node of nodes) {
    const requiredInputs = nodeIOConfig[node.type].inputs
    const hasInputs = edges.filter(e => e.target === node.id)
    
    if (requiredInputs.length > 0 && hasInputs.length === 0) {
      errors.push(`节点 "${node.id}" (${node.type}) 缺少输入连接`)
    }
  }
  
  // 检查是否有孤立节点
  // 检查循环依赖
  // ...
  
  return {
    valid: errors.length === 0,
    errors,
  }
}
```

```typescript
// apps/web/src/lib/workflow-dsl.ts
import { WorkflowDefinition } from '@/hooks/use-workflows'

// React Flow → 后端 DSL
export function toBackendDSL(flowNodes: any[], flowEdges: any[]): WorkflowDefinition {
  return {
    nodes: flowNodes.map(node => ({
      id: node.id,
      type: node.type as WorkflowNodeType,
      config: node.data.config || {},
      position: node.position,
    })),
    edges: flowEdges.map(edge => ({
      id: edge.id,
      source: edge.source,
      target: edge.target,
      condition: edge.data?.condition,
    })),
  }
}

// 后端 DSL → React Flow
export function fromBackendDSL(definition: WorkflowDefinition): { nodes: any[], edges: any[] } {
  return {
    nodes: definition.nodes.map(node => ({
      id: node.id,
      type: node.type,
      position: node.position || { x: 0, y: 0 },
      data: {
        name: node.config.name || node.type,
        config: node.config,
      },
    })),
    edges: definition.edges.map(edge => ({
      id: edge.id,
      source: edge.source,
      target: edge.target,
      data: { condition: edge.condition },
    })),
  }
}
```

```typescript
import debounce from 'lodash/debounce'

export function useAutoSave() {
  const { nodes, edges } = useCanvasStore()

  useEffect(() => {
```

## ⚠️ 风险与缓解

### 风险 1: React Flow 学习曲线

**影响**: Agent A/B 进度延迟

**缓解**:
- ✅ 提前阅读官方文档（https://reactflow.dev/docs）
- ✅ 从官方示例开始（基础→高级）
- ✅ 第一周每天预留 1 小时学习时间

### 风险 2: 状态管理复杂度

**影响**: Zustand store 过于复杂，难以维护

**缓解**:
- ✅ 按模块拆分 store（nodes、edges、viewport）
- ✅ 使用中间件（persist、devtools）
- ✅ 编写 store 使用文档

### 风险 3: 性能问题

**影响**: 大工作流渲染卡顿

**缓解**:
- ✅ 使用 React Flow 内置虚拟滚动
- ✅ 节点懒加载（可视区域外不渲染）
- ✅ 防抖自动保存（1 秒延迟）

### 风险 4: 内存压力（M4 16GB）

**影响**: 同时运行多个服务时系统卡顿

**缓解**:
- ✅ 分时运行 Ollama（开发前端时停止）
- ✅ 限制 Next.js 内存使用
- ✅ 监控内存，超过 12GB 时告警
